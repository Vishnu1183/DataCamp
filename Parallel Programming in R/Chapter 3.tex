################################# foreach, future.apply and Load Balancing ###############################


##### Combining results
# Load the package
library(foreach)

# foreach() %do% construct
result <- foreach(let = letters, .combine = c) %do% 
                max_frequency(let, words = words, min_length = 5)
            
# Plot results
barplot(result, las = 2)
